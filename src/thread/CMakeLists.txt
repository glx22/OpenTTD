cmake_minimum_required(VERSION 3.5)

# Threading
add_files(thread.h)
if (OPTION_USE_THREADS)
    if (WIN32)
        add_files(thread_win32.cpp)
    else (WIN32)
        if (OPTION_OS2)
            add_files(thread_os2.cpp)
        else (OPTION_OS2)
            if (OPTION_MORPHOS)
                add_files(thread_morphos.cpp)
            else (OPTION_MORPHOS)
                add_files(thread_pthread.cpp)
            endif (OPTION_MORPHOS)
        endif (OPTION_OS2)
    endif (WIN32)
else (OPTION_USE_THREADS)
    add_files(thread_none.cpp)
endif (OPTION_USE_THREADS)
